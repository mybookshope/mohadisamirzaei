 const cartItems = document.getElementById("cartItems");
 const addToCartButtons = document.querySelectorAll(".add-to-cart");
 addToCartButtons.forEach(btn => {
   btn.addEventListener("click", function() {
     const bookTitle = this.closest(".card").querySelector(".card-title").innerText;
     const bookPrice = this.closest(".card").querySelector(".fw-bold").innerText;
     const li = document.createElement("li");
     li.className = "list-group-item";
     li.innerText = `${bookTitle} - ${bookPrice}`;
     cartItems.appendChild(li);
   });
 });
 document.getElementById("checkoutBtn").addEventListener("click", function() {
   alert("Your payment was successful!");
   cartItems.innerHTML = '<li class="list-group-item">Your cart is empty</li>';
 });
